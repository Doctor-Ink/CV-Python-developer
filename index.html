          def rev_rot(strng, sz):
            result_str = []
            if sz == 0:
                return ''
            for x in range(0, int(len(strng)/sz)):
                chunk = strng[x * sz : (x + 1) * sz]
                if sum([int(x) ** 3 for x in chunk]) % 2 == 0:
                    result_str += chunk[::-1]
                elif sum([int(x) ** 3 for x in chunk]) % 2 != 0:
                    result_str += chunk[1::] + chunk[:1]
            return ''.join(result_str)
